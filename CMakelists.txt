cmake_minimum_required(VERSION 3.7.1)
project(FISC-VM)

	FVMAPILinux 
	FVMAPIWindows 
	FVMGenDebug
	FVMPass
	FVMRegistry
	FVMRuntime

	########## Support Libraries ##########

	FVMFISCISA

	FVMFISCPassPipeline
